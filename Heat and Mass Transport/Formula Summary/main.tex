%==========Heat Transport==========%
\begin{mdframed}\vspace{-.5cm}
\paragraph{Heat Transport}
\begin{multicols}{2}
    \begin{itemize}
    \item \textbf{Conduction}, Fourier's Law: $\displaystyle \dot{Q} = -kA\dfrac{\mathrm{d}T}{\mathrm{d}x}$
    
    \item \textbf{Convection}, Newton's Law of cooling: $\displaystyle \dot{Q} = hA (T_{s} - T_{\infty})$
    
    \item \textbf{Reynolds Transport Theorem}: 
    \[
        \frac{\mathrm{d}B_{sys}}{\mathrm{d}t}=\frac{\partial}{\partial t}\int\limits_{\mathrm{CV}}\rho \beta \mathrm{d}V+\oint\limits_{\mathrm{CS}}\rho\beta(\mathbf{v}\cdot\mathbf{n})\mathrm{d}A
    \]
    where in heat transfer, $B \equiv \text{thermal energy}$, $\beta \equiv c_p T$.
    
    \item Integral form of heat equation: 
    \[\hspace{-1cm}
        \underbrace{\frac{\partial}{\partial t} \int\limits_{\mathrm{CV}} \rho c_{p} T \mathrm{d}V}_{\substack{\text{rate of change}\\\text{of thermal energy}}} 
        = \underbrace{\int\limits_{\mathrm{CV}} \dot{S}_{v} \mathrm{d}V}_{\substack{\text{rate of heat}\\ \text{generation}}} 
        - \underbrace{\oint\limits_{\mathrm{CS}} (\mathbf{q} \cdot \mathbf{n}) \mathrm{d}A}_{\substack{\text{rate of heat loss}\\\text{due to conduction}}} 
        - \underbrace{\oint\limits_{\mathrm{CS}} \rho c_{p} T (\mathbf{v} \cdot \mathbf{n}) \mathrm{d}A}_{\substack{\text{rate of heat loss}\\ \text{due to convection}}} 
        - \dot{W}
    \]

    \item Differential form of heat equation: \textcolor{teal}{Lagrangian} = \textcolor{purple}{Eulerian}; 
    \[ 
        {\color{teal}
            \underbrace{\rho c_{p} \left( \frac{\partial T}{\partial t} +(\mathbf{v} \cdot \nabla)  T\right)}_{\substack{\text{rate of heat change}\\\text{in a fluid particle}}}
        } 
        = 
        {\color{purple}
            \underbrace{\dot{S}_{v}}_{\substack{\text{heat} \\ \text{generation}}}  + \underbrace{k \nabla^{2} T}_{\substack{\text{heat diffusion} \\ \text{ by conduction}}}
        }
    \]
\end{itemize}
\end{multicols}
\end{mdframed}
%==========Heat Conduction==========%

\begin{mdframed}\vspace{-.5cm}
\paragraph{Conduction}
\begin{multicols}{2}

When $\Bi \ll 1 \ \Rightarrow \ R_{\rm condunction} \ll R_{\rm convection}$, temperature distribution inside the solid is approximately uniform:
\[
    \dot{Q} = \Delta T / R_{T}.
\]

\begin{table}[H] \vspace{-.5cm}
    \centering
    \begin{tabular}{lc}
    \toprule
    & \textbf{Thermal Resistance, $R_{T}(\mathrm{K}/\mathrm{W})$}\\ 
    \midrule
    Conduction (planar)  & $L/kA$\\[2ex]
    Conduction (cylindrical) & $\displaystyle \frac{\ln(r_{2}/r_{1})}{2\pi Lk}$\\[2ex]
    Conduction (spherical)  & $\displaystyle \frac{1}{4\pi k} \left(\frac{1}{r_{1}}-\frac{1}{r_{2}}\right)$\\[2ex] 
    Convection &  $1/hA$\\
    \bottomrule
    \end{tabular}
\end{table} 

\begin{itemize}
    \item Lumped capacitance method: 
    \[
        \displaystyle T(t) =( T_{i} - T_{\infty} ) \ \exp\left( -\frac{t}{\tau} \right) + T_{\infty} 
        \quad \text{where} \ 
        \tau=\frac{\rho c_{p} V}{hA}.
    \]

    \item Transient conduction through semi-infinite solid:
	\begin{itemize}
            \item Penetration depth of the ``front'': $x  \sim  \sqrt{\alpha t}$
            
            \item Error function: $\displaystyle T(x, t) = (T_{s}-T_{i}) \ \mathrm{erfc}\bigg(\frac{x}{2\sqrt{\alpha t}}\bigg)+T_{i}$.
            
            \item Heat flux at the surface: $\displaystyle q_{s} = \sqrt{\frac{k \rho c_{p}}{\pi t}}(T_{s}-T_{i})$

            \item Interfacial temperature:
            \[\hspace{-1cm}
                T_{s}=\frac{T_{A}m_{A}+T_{B}m_{B}}{m_{A}+m_{B}}, \quad m_{A} = \sqrt{k_{A}\rho_{A}c_{p,A}}, \quad m_{B} = \sqrt{k_{B}\rho_{B}c_{p,B}}
            \]
	\end{itemize}
\end{itemize}
\end{multicols}
\end{mdframed}
%==========Heat Convection==========%
\begin{mdframed} \vspace{-.5cm}
\paragraph{Convection}

\begin{multicols}{2}
\begin{itemize}    
    \item Velocity boundary layer thickness: 
    \[
        \delta_{v} \sim \sqrt{\nu t} \sim \sqrt{\nu x/u_{\infty}}
        \quad \text{where} \
        \nu=\mu/\rho.
    \]
    
    \item Thermal boundary layer thickness: 
    \[
        \delta_{T} \sim \sqrt{\alpha t} \sim \sqrt{\alpha x/u_{\infty}}
        \quad \text{where} \
        \alpha = k/\rho c_p.
    \]
    
    \item Given a thermal boundary layer, to find $h$: 
    \[
        h = - \frac{k}{T_{s}-T_{\infty}} \frac{\mathrm{d}T}{\mathrm{d}y} \bigg \vert_{y=0} 
    \]
    
    \item Reynolds number: $\displaystyle \Ree_{x} = \frac{\rho u_{\infty} x}{\mu}=\frac{u_{\infty}x}{\nu}$: laminar/turbulent?

    \item Prandtl number: $\Prr = \nu/\alpha$: compare $\delta_v$ to $\delta_T$.

\end{itemize}
\end{multicols}

\begin{itemize}
    \item The average convective coefficient: $\displaystyle \overline{h} = \frac{1}{L} \int_{0}^L h \ \mathrm{d}x = \frac{1}{L} \left( \int_{0}^{x_c} h_{\rm lam} \ \mathrm{d}x + \int_{x_c}^{L} h_{\rm turb} \ \mathrm{d}x \right)$, where $x_c = \Ree_x \dfrac{\nu}{u_\infty}$ is the critical length.
\end{itemize}

\textbf{Empirical Nusselt number correlations, $\Nuu(\Ree, \Prr)$, for convective heat transfer:}
\begin{itemize}   
    \item \underline{Laminar flow} over an \underline{isothermal flat plate}: $\Ree_{x} \ll 5 \times 10^{5}$:
    $ \displaystyle
        \begin{cases}
            \text{when} \ \Prr \ll 1
            &
            \Nuu_{x} = 0.564 \ \Ree_{x}^{\frac{1}{2}} \ \Prr^{\frac{1}{2}}
            , \quad 
            \overline{\Nuu}_{L} = 1.128 \ \Ree_{L}^{\frac{1}{2}} \ \Prr^{\frac{1}{2}}\\
            
            \text{when} \ \Prr > 0.5
            &
            \Nuu_{x} = 0.332 \ \Ree_{x}^{\frac{1}{2}} \ \Prr^{\frac{1}{3}}
            , \quad
            \overline{\Nuu}_{L} = 0.664 \ \Ree_{L}^{\frac{1}{2}} \ \Prr^{\frac{1}{3}}
        \end{cases}
    $
    
    \item \underline{Turbulent flow} over an \underline{isothermal flat plate}: 
    \begin{center}
    when
    $
        \begin{cases}
            5 \times 10^{5}< \Ree_{x} < 3 \times 10^{7} \\
            0.7 < \Prr < 400
        \end{cases}
    $
    :
    $ \displaystyle
        \begin{cases}
            \Nuu_{x} = 0.029 \ \Ree_{x}^{0.8} \ \Prr^{0.43} \\
            \overline{\Nuu}_{L} = 0.664 \ \Ree_{c}^{\frac{1}{2}} \ \Prr^{\frac{1}{3}} + 0.036 \ \Ree_{L}^{0.8} \ \Prr^{0.43} \left[ 1- \left( \dfrac{\Ree_{c}}{\Ree_{L}} \right)^{0.8} \right]
        \end{cases}
    $
    \end{center}
    
    \item \underline{Laminar, fully developed} flow through a \underline{circular pipe}: $\Ree_{D} \ll 2300$:
    $ \displaystyle
        \begin{cases}
            \Nuu_{D} = 4.36 & \text{Uniform surface heat flux (uniform $q_{h}$)} \\
            \Nuu_{D} = 3.66 & \text{Uniform surface temperature (uniform $T_{s}$)}
        \end{cases}
    $

    \item \underline{Turbulent flow, fully developed} flow through a \underline{circular pipe}: when $\Ree_{D} > 10^{4}$, $\Prr > 0.7$: $\Nuu_{D} = 0.023 \ \Ree_{D}^{0.8} \ \Prr^{0.4}$.
\end{itemize}

\vspace{.2cm}
\textbf{Internal flow, velocity-averaged temperature characteristics:}
\begin{itemize}
    \item Velocity-averaged temperature in a circular pipe (radius $R$): $\displaystyle T_{m} = \frac{2\pi}{\dot{V}} \int_{0}^{R} T(r) u(r) r \ \mathrm{d}r$, where $\dot{V} = \pi \overline{u} R^2$ is the volumetric flow rate for fully-developed velocity profile $[\rm m^3/s]$.

    \item The temperature gradient along $x$-direction: $\displaystyle \frac{\partial T_m}{\partial x} = \frac{\pi D}{\dot{m} c_p} h (T_s - T_m)$, where $\dot{m} = \rho \overline{u} \pi R^2$ is the mass flow rate $[\rm kg/s]$.

    \item (constant heat flux) The temperature distribution along $x$-direction: $\displaystyle T_m(x) = \frac{\pi D \cdot q_s}{\dot{m} c_p} \cdot x + T_{m, {\rm in}}$.

    \item (constant heat flux) The convective heat flow rate: $\displaystyle \dot{Q} = \dot{m} c_p (T_{m,{\rm out}} - T_{m, {\rm in}})$.

    \item (isothermal surface) The temperature distribution: $\displaystyle \frac{\Delta T_{\rm out}}{\Delta T_{\rm in}} = \exp\left( - \frac{A \overline{h}}{\dot{m} c_p} \right)$, where $\Delta T_{(\star)} = T_{s,(\star)} - T_{m,(\star)}$.

    \item (isothermal surface) The convective heat flow rate: $\displaystyle \dot{Q} = \overline{h} A \Delta T_{lm}$, where $\displaystyle \Delta T_{lm} = \frac{\Delta T_{\rm out} - \Delta T_{\rm in}}{\ln (\Delta T_{\rm out}/\Delta T_{\rm in})}$.
\end{itemize}
\end{mdframed}

%==========Heat Convection==========%
\begin{mdframed} \vspace{-.5cm}
\paragraph{Mass Transport}

\begin{itemize}
\item Two modes of mass transport:

    \begin{center}
    \begin{tabular}{ccc}
    \toprule
     & Mass flow rate ($\mathrm{kg/s}$) & Mass transfer resistance ($\rm kg/m^3$) \\ 
    \midrule
    Advection 
    & $\dot{J}_A = v c A$ 
    & $R = 1/k_m A$ \\[1.5ex]
    
    Diffusion 
    & $\dot{J}_D = - \mathcal{D} A \dfrac{\mathrm{d}c}{\mathrm{d}x}$ 
    & $R = L/\mathcal{D} A$ \\
    
    \bottomrule
    \end{tabular}
    \end{center}
    
    \begin{itemize}
        \item Total mass transport \underline{flux}: $j_{\text{total}} = j_D + j_A = -\mathcal{D} \nabla c + vc$.
    
        \item Diffusivity (Stokes--Einstein relation): $\mathcal{D} = \dfrac{k_B T}{6\pi \mu a}$, where the particle radius $a = \left(\dfrac{3M_w}{4\pi \rho N_A}\right)^{1/3}$.
    \end{itemize}

    \item Integral form of conservation of mass solute:
    $ \displaystyle
        \frac{\partial}{\partial t} \int\limits_{\mathrm{CV}} c \mathrm{d}V = \int\limits_{\mathrm{CV}} \dot{S_{v}} \mathrm{d}V - \oint\limits_{\mathrm{CS}} (\mathbf{j} \cdot \mathbf{n}) \mathrm{d}A - \oint\limits_{\mathrm{CS}} c (\mathbf{v} \cdot \mathbf{n}) \mathrm{d}A 
    $
    
    \item Differential form of conservation of mass solute:
    $ \displaystyle
        \frac{\partial c}{\partial t} + (\mathbf{v} \cdot \nabla)c = \mathcal{D}\nabla^2 c + \dot{S}_{v}
    $
\end{itemize}

\vspace{.2cm}
\textbf{Mass transfer (concentration) boundary layer:}
\begin{itemize}
    \item Velocity boundary layer thickness: $\delta_{c} \sim \sqrt{\mathcal{D} t} \sim \sqrt{\mathcal{D} x/u_{\infty}}$.

    \item Empirical Sherwood number correlations, $\Sh(\Ree, \Sc)$, for convective heat transfer:
    \begin{itemize}
        \item Laminar flow over a flat plate: when
        $
            \begin{cases}
                \Ree_{x} < 5 \times 10^{5}\\
                0.6 < \Sc < 50
            \end{cases}
        $
        :
        $ \displaystyle
            \begin{cases}
                \Sh = 0.332 \ \Ree_{x}^{\frac{1}{2}} \ \Sc^{\frac{1}{3}} \\
                \overline{\Sh} = 0.664 \ \Ree_{x}^{\frac{1}{2}} \ \Sc^{\frac{1}{3}}
            \end{cases}
        $
    
        \item Turbulent flow over a flat plate: when
        $
            \begin{cases}
                5 \times 10^{5}< \Ree_{x} < 1 \times 10^{8} \\
                0.6 < \Sc < 50
            \end{cases}
        $
        :
        $ \displaystyle
            \begin{cases}
                \Sh = 0.0296 \ \Ree_{x}^{\frac{4}{5}} \ \Sc^{\frac{1}{3}} \\
                \overline{\Sh} = \left( 0.037 \ \Ree_{x}^{\frac{4}{5}} - 871 \right)\ \Sc^{\frac{1}{3}} 
            \end{cases}
        $
    \end{itemize}
\end{itemize}
